#!/data/data/com.termux/files/usr/bin/bash

export TERM=xterm-256color

pkg install -y android-tools coreutils grep sed gawk >/dev/null 2>&1

BAR_WIDTH=30
TOTAL_STEPS=100
STEP=0

ESC="\033"
RST="${ESC}[0m"

C_HDR="${ESC}[93m"
C_TXT="${ESC}[97m"
C_DIM="${ESC}[90m"
C_BAR="${ESC}[96m"
C_FILL="${ESC}[92m"
C_RED="${ESC}[91m"
C_CYAN="${ESC}[36m"
C_ORN="${ESC}[38;5;202m"

print_ui() {

PCT=$((STEP * 100 / TOTAL_STEPS))

[ $PCT -lt 0 ] && PCT=0
[ $PCT -gt 100 ] && PCT=100

FILLED=$((BAR_WIDTH * STEP / TOTAL_STEPS))

[ $FILLED -lt 0 ] && FILLED=0
[ $FILLED -gt $BAR_WIDTH ] && FILLED=$BAR_WIDTH

BAR=""

for ((i=1;i<=BAR_WIDTH;i++)); do
    if [ $i -le $FILLED ]; then
        BAR="${BAR}#"
    else
        BAR="${BAR}-"
    fi
done

printf "\033c"

echo -e "${C_HDR}╔════════════════════════════════════════════════════╗${RST}"
echo -e "${C_HDR}║${RST}              ${C_ORN}CÀI ĐẶT TV XIAOMI${RST}                ${C_HDR}║${RST}"
echo -e "${C_HDR}║${RST}          ${C_BAR}Việt Nguyễn - 0343.22.08.93${RST}          ${C_HDR}║${RST}"
echo -e "${C_HDR}╟────────────────────────────────────────────────────╢${RST}"

echo -e " ${C_BAR}[ ${C_FILL}${BAR}${RST}${C_BAR} ] ${C_TXT}${PCT}%${RST}"

echo -e "${C_HDR}╚════════════════════════════════════════════════════╝${RST}"

echo ""
}

tick() {

STEP=$((STEP + 1))

[ $STEP -gt $TOTAL_STEPS ] && STEP=$TOTAL_STEPS

print_ui
}

termux-setup-storage >/dev/null 2>&1

while true; do

STEP=0
IP_OK=0

while [ $IP_OK -eq 0 ]; do

printf "\033c"

echo -e "${C_HDR}╔═══════════════════════════════════════════════════╗${RST}"
echo -e "${C_HDR}║${RST}               ${C_ORN}CÀI ĐẶT TV XIAOMI${RST}               ${C_HDR}║${RST}"
echo -e "${C_HDR}║${RST}         ${C_BAR}Việt Nguyễn - 0343.22.08.93${RST}         ${C_HDR}║${RST}"
echo -e "${C_HDR}╚═══════════════════════════════════════════════════╝${RST}"

echo ""

printf "${C_CYAN}➤ Nhập IPv4 của TV: ${RST}"

IFS= read -r RAW_IP

IP=$(echo "$RAW_IP" | tr -d ' ')

if [ -z "$IP" ]; then
    echo -e "${C_RED}[!] Không được để trống IP${RST}"
    sleep 1
    continue
fi

echo -ne "${C_DIM}Đang kết nối tới ${IP}...${RST}"

adb kill-server >/dev/null 2>&1
adb start-server >/dev/null 2>&1

CONNECTED=0
COUNT=0

while [ $COUNT -lt 10 ]; do

    adb connect "$IP:5555" >/dev/null 2>&1

    if adb devices | grep "$IP:5555" | grep "device" >/dev/null 2>&1; then
        CONNECTED=1
        break
    fi

    echo -n "."
    sleep 1

    COUNT=$((COUNT + 1))
done

if [ $CONNECTED -eq 1 ]; then

    echo -e "\n${C_FILL}[✓] Kết nối thành công${RST}"

    IP_OK=1

    sleep 1

else

    echo -e "\n${C_RED}[!] Không thể kết nối tới TV${RST}"
    echo -e "${C_TXT}Hãy bật ADB Debugging trên TV${RST}"

    echo ""
    printf "${C_ORN}Nhấn Enter để thử lại...${RST}"

    IFS= read -r
fi

done

MAN=$(adb shell getprop ro.product.manufacturer | tr -d '\r')
MOD=$(adb shell getprop ro.product.model | tr -d '\r')
COD=$(adb shell getprop ro.product.name | tr -d '\r')
SER=$(adb shell getprop ro.serialno | tr -d '\r')
BUI=$(adb shell getprop ro.build.display.id | tr -d '\r')
SEC=$(adb shell getprop ro.build.version.security_patch | tr -d '\r')
AND=$(adb shell getprop ro.build.version.release | tr -d '\r')
SDK=$(adb shell getprop ro.build.version.sdk | tr -d '\r')

printf "\033c"

echo -e "${C_HDR}╔══════════════════════════════════════════════════╗${RST}"
echo -e "${C_HDR}║${RST}            ${C_ORN}THÔNG TIN THIẾT BỊ${RST}              ${C_HDR}║${RST}"
echo -e "${C_HDR}╚══════════════════════════════════════════════════╝${RST}"

echo ""
echo -e "Hãng SX     : ${C_FILL}${MAN}${RST}"
echo -e "Model       : ${C_FILL}${MOD}${RST}"
echo -e "Code        : ${C_FILL}${COD}${RST}"
echo -e "Serial      : ${C_FILL}${SER}${RST}"
echo -e "Build       : ${C_FILL}${BUI}${RST}"
echo -e "Security    : ${C_FILL}${SEC}${RST}"
echo -e "Android     : ${C_FILL}${AND} (SDK ${SDK})${RST}"

IS_2026=0

echo "$MOD" | grep -iq "2026" && IS_2026=1

[ "$SDK" -ge 34 ] && IS_2026=1

if [ "$IS_2026" = "1" ]; then
    TOTAL_STEPS=79
    MSG="XIAOMI TV 2026 - HyperOS 3"
else
    TOTAL_STEPS=63
    MSG="XIAOMI TV 2025 - MIUI"
fi

echo ""
echo -e "Phát hiện : ${C_ORN}${MSG}${RST}"

echo ""
printf "${C_RED}➤ Nhấn Enter để bắt đầu cài đặt...${RST}"

IFS= read -r

tick

adb shell svc power stayon true >/dev/null 2>&1; tick

adb shell settings put global device_locales vi-VN >/dev/null 2>&1; tick
adb shell settings put global sys_locale vi-VN >/dev/null 2>&1; tick
adb shell settings put system system_locales vi-VN >/dev/null 2>&1; tick

adb shell settings put global animator_duration_scale 0 >/dev/null 2>&1; tick
adb shell settings put global transition_animation_scale 0 >/dev/null 2>&1; tick
adb shell settings put global window_animation_scale 0 >/dev/null 2>&1; tick

adb shell settings put global stay_on_while_plugged_in 3 >/dev/null 2>&1; tick

APK_LIST=(
466.apk
470.apk
mixp.apk
voice.apk
keyboard.apk
ytb.apk
quantv.apk
sportzx.apk
mstore.apk
ferrari.apk
DL.apk
getout.apk
mapvoice.apk
phim4k.apk
)

for apk in "${APK_LIST[@]}"; do

if [ -f "$apk" ]; then
    adb install -g "$apk" >/dev/null 2>&1
fi

tick

done

adb shell ime enable com.liskovsoft.leankeyboard/.ime.LeanbackImeService >/dev/null 2>&1; tick

adb shell settings put secure default_input_method com.liskovsoft.leankeyboard/.ime.LeanbackImeService >/dev/null 2>&1; tick

adb reboot >/dev/null 2>&1 &

sleep 8

echo ""
echo -e "${C_ORN}Đang chờ TV khởi động lại...${RST}"

RETRY=0

while [ $RETRY -lt 30 ]; do

adb connect "$IP:5555" >/dev/null 2>&1

STATE=$(adb -s "$IP:5555" get-state 2>/dev/null)

if [ "$STATE" = "device" ]; then

    echo -e "${C_FILL}[✓] TV đã online${RST}"
    break
fi

echo -n "."

sleep 5

RETRY=$((RETRY + 1))

done

tick

adb shell svc power stayon false >/dev/null 2>&1; tick

adb kill-server >/dev/null 2>&1; tick

print_ui

echo ""
echo -e "${C_FILL}✅ CÀI ĐẶT HOÀN TẤT${RST}"

echo ""
echo -e "${C_CYAN}1.${RST} Cài cho TV khác"
echo -e "${C_CYAN}2.${RST} Thoát"

echo ""

printf "${C_ORN}➤ Chọn: ${RST}"

IFS= read -r OPT

if [ "$OPT" != "1" ]; then

    adb kill-server >/dev/null 2>&1

    echo ""
    echo -e "${C_RED}Đã thoát chương trình${RST}"

    exit 0
fi

done
