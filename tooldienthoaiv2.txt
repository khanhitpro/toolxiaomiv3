#!/bin/bash

BAR_WIDTH=30
TOTAL_STEPS=100
STEP=0
ESC="\033"
RST="${ESC}[0m"
C_HDR="${ESC}[93m"
C_TXT="${ESC}[97m"
C_DIM="${ESC}[90m"
C_BAR="${ESC}[96m"
C_FILL="${ESC}[92m"
C_RED="${ESC}[91m"
C_CYAN="${ESC}[36m"
C_ORN="${ESC}[38;5;202m"
RED='\033[1;31m'
GREEN='\033[1;32m'
YELLOW='\033[1;33m'
BLUE='\033[1;34m'
CYAN='\033[1;36m'
WHITE='\033[1;37m'
MAGENTA='\033[1;35m'
NC='\033[0m'

print_ui() {
  PCT=$((STEP * 100 / TOTAL_STEPS))
  [ $PCT -lt 0 ] && PCT=0
  [ $PCT -gt 100 ] && PCT=100

  FILLED=$((BAR_WIDTH * STEP / TOTAL_STEPS))
  [ $FILLED -lt 0 ] && FILLED=0
  [ $FILLED -gt $BAR_WIDTH ] && FILLED=$BAR_WIDTH

  BAR=""
  for ((i=1;i<=BAR_WIDTH;i++)); do
    if [ $i -le $FILLED ]; then
      BAR="${BAR}█"
    else
      BAR="${BAR}░"
    fi
  done

  echo -e "${BLUE}📡 MENU KẾT NỐI TIVI XIAOMI PRO${NC}"
  echo -e "${WHITE}👨‍💻 Khánh IT | ☎ 0967.341.608${NC}"

  echo -e "${WHITE}⚠️ Bản Cập Nhật: 01-05-2026${NC}"
  echo ""
  echo -e "  ${C_BAR}[ ${C_FILL}${BAR}${RST}${C_BAR} ] ${C_TXT}${PCT}%${RST} ${C_DIM}(Bước ${STEP}/${TOTAL_STEPS})${RST}"
  echo ""
  echo -e "${MAGENTA}📺 HƯỚNG DẪN KẾT NỐI TIVI${NC}"
  echo -e "${CYAN}🎮 Điều khiển:${NC} Nhấn ${YELLOW}↑ lên trên cùng${NC} → vào ${WHITE}Cài đặt ⚙${NC}"

  echo -e "${CYAN}🔓 Developer:${NC} Vào ${WHITE}About${NC} → chọn ${YELLOW}Model${NC} → ${GREEN}OK 9 lần${NC}"
  echo -e "${CYAN}⛔ Lưu ý:${NC} Tắt ${RED}Update tự động${NC}"
  echo -e "${CYAN}⚙️ Quyền:${NC} Account - Security → Bật ${GREEN}ADB Debugging${NC} + ${GREEN}Unknown sources${NC}"
  echo -e "${CYAN}📶 Kết nối:${NC} Bật ${WHITE}WiFi${NC} - Lấy địa chỉ IP : ${ip}"
  echo ""
  echo -e "${C_CYAN} Báo lỗi vui lòng liên hệ:${RST} ${C_ORN}0967341608${RST}"
  echo ""
}

tick() {
  STEP=$((STEP+1))
