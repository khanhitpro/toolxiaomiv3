# ===== THÔNG SỐ TIVI CHẠY TỰ DỘNG KHI KẾT NỐI THÀNH CÔNG =====
get_tv_info() {
  MODEL=$(adb shell getprop ro.product.model | tr -d '\r')
  BRAND=$(adb shell getprop ro.product.brand | tr -d '\r')
  PANEL=$(adb shell getprop ro.boot.mi.panel_size 2>/dev/null | tr -d '\r')
  ANDROID=$(adb shell getprop ro.build.version.release | tr -d '\r')
  PATCH=$(adb shell getprop ro.build.version.security_patch | tr -d '\r')
  BUILD_SHOW=$(adb shell getprop ro.build.version.incremental | tr -d '\r')
  DATE=$(adb shell getprop ro.build.date | tr -d '\r')
  SERIAL=$(adb shell getprop ro.serialno | tr -d '\r')
  MANUFACTURER=$(adb shell getprop ro.product.manufacturer | tr -d '\r')
  DEVICE=$(adb shell getprop ro.product.device | tr -d '\r')
  BOARD=$(adb shell getprop ro.product.board | tr -d '\r')

  # Nếu không có PANEL thì gán mặc định
  [ -z "$PANEL" ] && PANEL="?"

  echo -e "${CYAN}📺 Thông Tin Tivi: ${NC}${MODEL} ${GRAY}|${NC} ${YELLOW}${BRAND}${NC} ${GRAY}|${NC} ${GREEN}${PANEL} Inch${NC}"
  echo -e "${MAGENTA}🤖 Android ${NC}${ANDROID} ${GRAY}|${NC} ${BLUE}${BUILD_SHOW}${NC} ${GRAY}|${NC} ${RED}${PATCH}${NC}"
  echo -e "${CYAN}📅 Build Date:${NC} ${DATE} ${GRAY}|${NC} ${YELLOW}🔢 Serial:${NC} ${SERIAL}"
  echo -e "${GREEN}🏭 Manufacturer:${NC} ${MANUFACTURER} ${GRAY}|${NC} ${CYAN}📱 Device:${NC} ${DEVICE} ${GRAY}|${NC} ${MAGENTA}🧩 Board:${NC} ${BOARD}"
}


# ===== MENU KẾT NỐI TIVI CONNECT =====
menu_connect() {
  while true; do
    clear
    banner_menu1
    read -e -p $'\e[1;36m👉 Nhập IP (192.168.1.10): \e[0m' ip
    [ "$ip" = "0" ] && break

    adb connect "$ip:5555" >/dev/null 2>&1
    sleep 5

    if adb devices | grep -q "$ip:5555.*device"; then
      echo -e "${GREEN}✅ KẾT NỐI THÀNH CÔNG!${NC}"
      sleep 1
      get_tv_info
      menu_main
    else
      echo -e "${RED}❌ KẾT NỐI THẤT BẠI!${NC}"
      sleep 2
    fi
  done
}

# ===== DOWNLOAD (GITHUB JSON) =====
download_json() {
  DIR="$HOME/storage/shared/apktool"
  JSON_URL="https://raw.githubusercontent.com/khanhitxiaomi/toolxiaomi/main/apptoolxiaomi.json"

  mkdir -p "$DIR"
  clear

  echo "================================================"
  echo "🌐 ĐANG LOADING DANH SÁCH APP TỪ GITHUB........"
  echo "================================================"

  DATA=$(curl -s "$JSON_URL")

  if [ -z "$DATA" ]; then
    echo "❌ Không Loading Được Danh Sách App!"
    sleep 2
    return
  fi

  echo "✅ Loading Danh Sách App Thành Công"
  echo ""

  # 👉 parse JSON
  LIST=$(echo "$DATA" | grep -o '"name": *"[^"]*"\|"url": *"[^"]*"' \
  | sed 'N;s/\n/|/' \
  | sed 's/"name": *"//;s/"url": *"//;s/"//g')

  IFS=$'\n' read -d '' -r -a ARRAY <<< "$LIST"

  total=${#ARRAY[@]}

  if [ "$total" -eq 0 ]; then
    echo "❌ Không có dữ liệu trong JSON!"
    sleep 2
    return
  fi

  echo "📦 Tổng Số App: $total"
  echo "📂 Lưu tại: $DIR"
  echo "=============================================="
  echo ""

  count=0

  for item in "${ARRAY[@]}"; do
    count=$((count + 1))

    name=$(echo "$item" | cut -d'|' -f1)
    url=$(echo "$item" | cut -d'|' -f2)
    file="$DIR/$(basename "$url")"

    echo "📥 [$count/$total] $name"

    wget --progress=bar:force:noscroll "$url" -O "$file"

    if [ $? -eq 0 ]; then
      echo "✅ OK"
    else
      echo "❌ LỖI"
      rm -f "$file"
    fi

    echo ""
  done

  echo "==============================================="
  echo "🎉 TẢI APP HOÀN TẤT VUI LÒNG CHỌN CHẾ ĐỘ CÀI !"
  echo "==============================================="

  sleep 2
}

  echo -e "${CYAN}📦 Thành công: $ok | ❌ Lỗi: $fail${NC}"

  else
    echo "❌ Lỗi khi cài: p.apk"
  fi

  for apk in allapp.apk youtube.apk khanhstore-6.9.6.apk taive.apk imedia.apk vietplay.apk cotivi.apk; do

# ===== CÀI APK TRONG THƯ MỤC DOWNLOAD ĐIỆN THOẠI =====
install_from_phone_download() {
  DIR="$HOME/storage/downloads"

  echo -e "${CYAN}📂 Đang kiểm tra file APK trong: $DIR${NC}"
  echo ""

  # Nếu không có file .apk thì báo lỗi
  shopt -s nullglob
  files=("$DIR"/*.apk)
  if [ ${#files[@]} -eq 0 ]; then
    echo -e "${RED}❌ Không tìm thấy file APK nào trong Download${NC}"
    pause
    return
  fi

  count=0
  total=${#files[@]}

  for file in "${files[@]}"; do
    ((count++))
    echo -e "${YELLOW}[$count/$total] 📦 Đang cài: $(basename "$file")${NC}"
    if adb install -r -d "$file"; then
      echo -e "${GREEN}   ✅ Thành công${NC}"
    else
      echo -e "${RED}   ❌ Lỗi khi cài${NC}"
    fi
    echo ""
  done

  echo -e "${GREEN}🎉 Hoàn tất cài đặt tất cả file APK trong Download${NC}"
  pause
}



# ===== KHỞI ĐỘNG LẠI TIVI =====
reboot_tv() {
  echo -e "${BLUE}🔄 ĐANG KHỞI ĐỘNG LẠI TIVI...${NC}"
  echo ""

  # 👉 hiệu ứng chạy
  echo -n "⏳ Đang xử lý"
  for i in {1..3}; do
    echo -n "."
    sleep 0.5
  done
  echo ""

  # 👉 gửi lệnh reboot
  adb reboot

  echo -e "${GREEN}✅ Đã gửi lệnh reboot${NC}"
  echo ""

  # 👉 đếm ngược 10 giây
  echo -e "${YELLOW}⏱️ Đang chờ TV khởi động lại...${NC}"
  for i in $(seq 10 -1 1); do
    echo -ne "${WHITE}👉 $i giây...\r${NC}"
    sleep 1
  done

  echo ""
  echo -e "${GREEN}🎉 Hoàn tất!${NC}"
  echo -e "${CYAN}📡 Sau khi TV lên, hãy kết nối lại ADB${NC}"

  sleep 2
  pause
  menu_connect
}
